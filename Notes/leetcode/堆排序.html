#include <iostream>
#include <vector>
using namespace std;

void maxHeapify(vector<int>& arr, int n, int i); // 维护最大堆
void heapSort(vector<int>& arr); // 堆排序

// 主函数
int main() {
    vector<int> arr = {12, 11, 13, 5, 6, 7}; // 示例数组
    cout << "原始数组: ";
    for (int num : arr) {
        cout << num << " ";
    }
    cout << endl;

    heapSort(arr); // 调用堆排序函数

    cout << "排序后的数组: ";
    for (int num : arr) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}

// 维护最大堆
void maxHeapify(vector<int>& arr, int n, int i) {
    int largest = i; // 初始化最大值为当前节点
    int left = 2 * i + 1; // 左子节点
    int right = 2 * i + 2; // 右子节点

    // 如果左子节点大于当前最大值
    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }

    // 如果右子节点大于当前最大值
    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }

    // 如果最大值不是当前节点，交换并递归调整子树
    if (largest != i) {
        swap(arr[i], arr[largest]); // 交换当前节点和最大值节点
        maxHeapify(arr, n, largest); // 递归调整子树
    }
}

// 堆排序
void heapSort(vector<int>& arr) {
    int n = arr.size();

    // 构建最大堆
    for (int i = n / 2 - 1; i >= 0; i--) {
        maxHeapify(arr, n, i); // 从最后一个非叶子节点开始调整
    }

    // 逐一提取堆顶元素并调整堆
    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]); // 将堆顶元素（最大值）与最后一个元素交换
        maxHeapify(arr, i, 0); // 调整剩余的堆
    }
}
