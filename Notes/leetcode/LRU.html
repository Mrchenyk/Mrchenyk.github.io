class Node{
    public:
    int key,val;
    Node *pre,*next;
    Node(int k,int v){
        key=k,val=v;
    }
};

class LRUCache {
public:
    map<int,Node*>mp;
    
    Node* dummy;
    int capacity;
    LRUCache(int capacity) {
        this->capacity=capacity;
        dummy=new Node(0,0);
        dummy->pre=dummy;
        dummy->next=dummy;
    }
    
    int get(int key) {
        if(mp.contains(key)){
            remove(mp[key]);
            push_head(mp[key]);
            return mp[key]->val;
        }
        return -1;
    }
    
    void put(int key, int value) {
        
        if(mp.contains(key)){
            Node *node=mp[key];
            node->val=value;
            remove(node);
            push_head(node);
            
        }else{
            Node *node=new Node(key,value);
            push_head(node);
            mp[key]=node;
        }

        if(mp.size()>capacity){
            
            Node* tem=dummy->pre;
            cout<<tem->key<<endl;
            remove(tem);
            mp.erase(tem->key);
        }
    }

    void push_head(Node *node){
        node->pre=dummy;
        node->next=dummy->next;
        node->pre->next=node;
        node->next->pre=node;

    }

    void remove(Node *node){
        node->pre->next=node->next;
        node->next->pre=node->pre;
    }
};
